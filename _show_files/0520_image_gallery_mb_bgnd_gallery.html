JavaScript

$.mbBgndGallery.buildGallery({
  // containment:"#wrapper",
  containment:"body",
  effect:{enter:{top:"100%",opacity:1},exit:{top:"-100%",opacity:1}},
  timer:3000,
  effTimer:500,
  raster:"assets/raster.png", // ラスター画像を上に被せるとおしゃれ
  controls:"#controls",
  images:[
    "assets/gallery/0001.jpg",
    "assets/gallery/0002.jpg",
    "assets/gallery/0003.jpg",
    "assets/gallery/0004.jpg",
    "assets/gallery/0005.jpg",
    "assets/gallery/0006.jpg",
    "assets/gallery/0007.jpg",
    "assets/gallery/0008.jpg",
    "assets/gallery/0009.jpg",
    "assets/gallery/0010.jpg",
  ],
  autoStart:false,
  onStart:function(){$("#controls .play").show();$("#controls .pause").hide();},
  onPause:function(){$("#controls .play").show();$("#controls .pause").hide();},
  onPlay:function(o){$("#controls .play").hide();$("#controls .pause").show();o.effect={enter:{top:"100%",opacity:1},exit:{top:"-100%",opacity:1}}},
  onNext:function(o){o.effect={enter:{top:"100%",opacity:1},exit:{top:"-100%",opacity:1}}},
  onPrev:function(o){o.effect={enter:{top:"-100%",opacity:1},exit:{top:"100%",opacity:1}}}
});
