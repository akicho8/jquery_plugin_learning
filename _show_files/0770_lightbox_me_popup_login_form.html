68. lightbox_meでログインフォームをポップアップするには？

ログインする

HTML

<a href="#" id="do_login">ログインする</a>
<form id="main">
  <p>
    <label for="login">ログイン名</label>
    <input type="text" name="login" id="login" />
  </p>
  <p>
    <label for="password">パスワード</label>
    <input type="text" name="password" id="password" />
  </p>
  <p><input type="submit" value="ログイン" /></p>
</form>

JavaScript

$("#do_login").click(function() {
  $("#main").lightbox_me({
    centered: true, // 中央に表示
    overlayCSS: {background: "green", opacity: 0.5}, // まわりの輝度調整
    overlaySpeed: 125,  // 輝度が下がるスピード
    lightboxSpeed: 150, // フォームが出てくるスピード
    onLoad: function(){$("#main input:first").focus()}, // 最初の入力覧にフォーカス
  });
});
