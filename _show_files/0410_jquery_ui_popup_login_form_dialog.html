32. ポップアップでログインフォームを表示するには？

ログインする

HTML

<div id="main" title="ログイン">
  <form>
    <fieldset>
      <label for="login">ログイン名</label>
      <input type="text" name="login" id="login" value="" class="text ui-widget-content ui-corner-all" />
      <label for="password">パスワード</label>
      <input type="text" name="password" id="password" value="" class="text ui-widget-content ui-corner-all" />
    </fieldset>
  </form>
</div>
<a href="#" id="do_login">ログインする</a>

JavaScript

$("#dialog:ui-dialog").dialog("destroy");
$("#main").dialog({
  autoOpen: false, // 裏で起動しておくので display: none の設定が不要になる
  height: 300,
  width: 350,
  modal: true,
  buttons: {
     "ログイン": function() {
         $(this).dialog("close");
      }
  }});

$("#do_login").click(function() {
  $("#main").dialog("open");
});
